Install the below packages (apt install):

    qt5-default  (sudo apt-get update -y	sudo apt-get install -y qt5-default)
    qtdeclarative5-dev (sudo apt-get install -y qtdeclarative5-dev)
    qtquickcontrols2-5-dev (sudo apt-get -y install qtquickcontrols2-5-dev)
    libgles2-mesa-dev (sudo apt-get install -y libgles2-mesa-dev)
    qml-module-qtquick-controls (sudo apt-get install -y qml-module-qtquick-controls)
    qml-module-qtquick-extras (sudo apt-get install -y qml-module-qtquick-extras)
